BASE CASES

	INPUT: "abc123 abc 123 123.1"
	EXPECTED OUTPUT:
		word: abc123
		word: abc
		decimal integer: 123
		float: 123.1


	INPUT: "0x4356abdc 0777 [] "
	EXPECTED OUTPUT:
		hexadecimal integer: 0x4356abdc
		octal integer: 0777
		left bracket: [
		right bracket: ]

	INPUT: “array[xyz ] += pi 3.14159e-10”
	EXPECTED OUTPUT:
		word: array
		left bracket: [
		word: xyz
		right bracket: ]
		plus equals: +=
		word: pi
		float: 3.14159e-10

	INPUT: "if 123 char for"
	EXPECTED OUTPUT:
		c keyword: if
		decimal integer: 123
		c keyword: char
		c keyword: for

	INPUT: "<< >> >>= <<= != == = + - / *"
	EXPECTED OUTPUT:
		shift left: <<
		shift right: >>
		shift left equals: <<=
		shift right equals: >>=
		inequality: !=
		equality: ==
		assigment: =
		addition: +
		minus: -
		division: /
		multiple/dereference: *

	INPUT: ": ) ( [ ] ^ ^= && || += -= *= /="
	EXPECTED OUTPUT:
		conditional false: :
		right parenthesis: )
		left parenthesis: (
		left bracket: [
		right bracket: ]
		bitwise XOR: ^
		bitwise XOR equals: ^=
		logical AND: &&
		logical OR: ||
		plus equals: +=
		minus equals: -=
		times equals: *=
		divide equals: /=

	INPUT: ". -> sizeof , ! ++ --"
	EXPECTED OUTPUT:
		structure member: .
		structure pointer: ->
		c keyword: sizeof
		comma: ,
		negate: !
		increment: ++
		decrement: --
	
	INPUT: "~ ? 'test' "
	EXPECTED OUTPUT:
		1s compliment: ~
		conditional true: ?
		quotes: 'test'

	INPUT: "/* * hi */"
	EXPECTED OUTPUT:
		no output, since it will skip comments.

	INPUT: "//hi"
	EXPECTED OUTPUT:
		no output, since it will skip comments.

	INPUT: "123stuff"
	EXPECTED OUTPUT:
		decimal integer: 123
		word: stuff
